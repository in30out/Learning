# 导入方式
import 模块名                                      例：import random,os
	调用方式：模块名.功能名    例：random.randint(10,100)
import 模块名 as 别名                         例：import random as rd
	调用方式：别名.功能名        例：rd.randint(10,100)
from 模块名 import 功能名                 例：from random import randint,choice
	调用方式：功能名                例：randint(10,100)
from 模块名 import 功能名 as 别名    例：from random import randint as rint
	调用方式：别名                    例：rint(10,100)
from 模块名 import \*                          例：from random import \*
	调用方式：功能名                例：randint(10,100)

\_\_all\_\_ = \["函数名"，“变量名”\]     约束通配符\*

# 包
要配置__init__.py文件
from 包名 import  \*     要预先在\_\_init\_\_.py内配置\_\_all\_\_ = \["要导入的模块名"\]
